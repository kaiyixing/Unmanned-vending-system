# 多支付渠道实现方案

## 设计目标
- 支持多种支付方式（微信支付、支付宝、银联等）
- 策略模式实现，便于扩展
- 统一的支付回调处理
- 配置化管理，便于切换环境

---

## 支付架构设计

```java
package com.vending.module.payment.strategy;

import com.wechat.pay.java.core.Config;
import com.wechat.pay.java.core.RSAAutoCertificateConfig;
import com.wechat.pay.java.service.payments.jsapi.JsapiServiceExtension;
import com.wechat.pay.java.service.payments.jsapi.model.PrepayRequest;
import com.vending.module.order.entity.Order;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

import java.util.HashMap;
import java.util.Map;

@Component
public class WeChatPayStrategy implements PaymentStrategy {

    @Value("${payment.wechat.app-id}")
    private String appId;

    @Value("${payment.wechat.mch-id}")
    private String mchId;

    @Value("${payment.wechat.serial-no}")
    private String serialNo;

    @Value("${payment.wechat.api-v3-key}")
    private String apiV3Key;

    @Value("${payment.wechat.notify-url}")
    private String notifyUrl;

    @Value("${payment.wechat.private-key-path}")
    private String privateKeyPath;

    @Override
    public String getChannel() {
        return "WECHAT";
    }

    @Override
    public Map<String, Object> createPayment(Order order) throws Exception {
        Config config = new RSAAutoCertificateConfig.Builder()
                .merchantId(mchId)
                .privateKeyFromPath(privateKeyPath)
                .merchantSerialNumber(serialNo)
                .apiV3Key(apiV3Key)
                .build();

        JsapiServiceExtension service = new JsapiServiceExtension.Builder().config(config).build();

        PrepayRequest request = new PrepayRequest();
        request.setAppid(appId);
        request.setMchid(mchId);
        request.setDescription("智能货柜商品");
        request.setOutTradeNo(order.getOrderNo());
        request.setNotifyUrl(notifyUrl);
        request.setAmount(new com.wechat.pay.java.service.payments.jsapi.model.Amount());
        request.getAmount().setTotal(order.getTotalAmount().intValue());

        // 微信 JSAPI 支付
        Map<String, String> prepay = service.prepayWithRequestPayment(request);

        Map<String, Object> result = new HashMap<>();
        result.put("type", "wechat_jsapi");
        result.put("prepay", prepay);
        return result;
    }

    @Override
    public boolean processCallback(Map<String, String> params) throws Exception {
        // TODO: 验证微信签名
        return true;
```

### 5. 支付策略工厂

```java
// vending-server/src/main/java/com/vending/module/payment/factory/PaymentStrategyFactory.java

package com.vending.module.payment.factory;

import com.vending.module.payment.strategy.PaymentStrategy;
import jakarta.annotation.PostConstruct;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.HashMap;
import java.util.List;
import java.util.Map;

@Component
public class PaymentStrategyFactory {

    @Autowired
    private List<PaymentStrategy> strategies;

    private final Map<String, PaymentStrategy> strategyMap = new HashMap<>();

    @PostConstruct
    public void init() {
        strategies.forEach(s -> strategyMap.put(s.getChannel(), s));
    }

    public PaymentStrategy getStrategy(String channel) {
        PaymentStrategy strategy = strategyMap.get(channel.toUpperCase());
        if (strategy == null) {
            throw new IllegalArgumentException("不支持的支付方式: " + channel);
        }
        return strategy;
    }

    public List<String> getAvailableChannels() {
        return strategies.stream().map(PaymentStrategy::getChannel).toList();
    }
}
```

### 6. 更新支付服务

```java
// vending-server/src/main/java/com/vending/module/payment/service/impl/PaymentServiceImpl.java

package com.vending.module.payment.service.impl;

import com.baomidou.mybatisplus.extension.service.impl.ServiceImpl;
import com.vending.module.order.entity.Order;
import com.vending.module.payment.entity.PaymentRecord;
import com.vending.module.payment.factory.PaymentStrategyFactory;
import com.vending.module.payment.mapper.PaymentRecordMapper;
import com.vending.module.payment.service.PaymentService;
import com.vending.module.payment.strategy.PaymentStrategy;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.time.LocalDateTime;
import java.util.Map;
import java.util.UUID;

@Service
public class PaymentServiceImpl extends ServiceImpl<PaymentRecordMapper, PaymentRecord> implements PaymentService {

    @Autowired
    private PaymentStrategyFactory strategyFactory;

    @Override
    public Map<String, Object> createPayment(Order order, String channel) throws Exception {
        PaymentStrategy strategy = strategyFactory.getStrategy(channel);
        Map<String, Object> paymentData = strategy.createPayment(order);

        PaymentRecord record = new PaymentRecord();
        record.setOrderId(order.getOrderId());
        record.setOrderNo(order.getOrderNo());
        record.setChannel(channel);
        record.setAmount(order.getTotalAmount());
        record.setStatus(1);
        record.setTradeNo(paymentData.getOrDefault("tradeNo", "MOCK_" + UUID.randomUUID()).toString());
        record.setPayTime(LocalDateTime.now());

        this.save(record);

        return paymentData;
    }

    @Override
    public PaymentRecord createPaymentRecord(Order order, String channel) {
        PaymentRecord record = new PaymentRecord();
        record.setOrderId(order.getOrderId());
        record.setOrderNo(order.getOrderNo());
        record.setChannel(channel);
        record.setAmount(order.getTotalAmount());
        record.setStatus(1);
        record.setTradeNo("MOCK_" + UUID.randomUUID().toString().replace("-", ""));
        record.setPayTime(LocalDateTime.now());

        this.save(record);
        return record;
    }

    public boolean processPaymentCallback(String channel, Map<String, String> params) throws Exception {
        PaymentStrategy strategy = strategyFactory.getStrategy(channel);
        return strategy.processCallback(params);
    }
}
```

```vue
<template>

            >

            </div>
          </div>
        </div>

        </button>

<script setup>
import { ref, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { useCartStore } from '@/stores/cart'
import NavBar from '@/components/NavBar.vue'
import { createOrder } from '@/api/order'
import { createPayment } from '@/api/payment'

const router = useRouter()
const cartStore = useCartStore()

const orderNo = ref('')
const amount = ref(0)
const selectedChannel = ref('MOCK')

const paymentMethods = [
  { channel: 'WECHAT', name: '微信支付', icon: '💚' },
  { channel: 'ALIPAY', name: '支付宝', icon: '💙' },
  { channel: 'UNIONPAY', name: '银联支付', icon: '💳' },
  { channel: 'MOCK', name: '模拟支付', icon: '🎯' }
]

const selectChannel = (channel) => {
  selectedChannel.value = channel
}

const confirmPay = async () => {
  try {
    // 1. 创建订单
    const orderRes = await createOrder({
      cabinetId: cartStore.cabinetId,
      items: cartStore.items.map(i => ({
        productId: i.productId,
        quantity: i.quantity
      }))
    })

    orderNo.value = orderRes.data.orderNo
    amount.value = orderRes.data.totalAmount

    // 2. 调用支付
    const payRes = await createPayment({
      orderNo: orderNo.value,
      channel: selectedChannel.value
    })

    // 3. 处理支付响应
    handlePaymentResponse(payRes.data)
  } catch (e) {
    console.error(e)
  }
}

const handlePaymentResponse = (data) => {
  if (data.type === 'wechat_jsapi') {
    // 微信 JSAPI 支付
    window.WeChatPay.invoke(data.prepay, () => {
      cartStore.clearCart()
      router.push('/pickup/' + orderNo.value)
    })
  } else if (data.type === 'form') {
    // 支付宝表单提交
    const form = document.createElement('div')
    form.innerHTML = data.form
    document.body.appendChild(form)
    form.querySelector('form').submit()
  } else {
    // 模拟支付，直接跳转
    cartStore.clearCart()
    router.push('/pickup/' + orderNo.value)
  }
}

onMounted(() => {
  amount.value = cartStore.totalPrice
})
```

## 安全建议

1. **回调验签**：必须验证支付回调签名
2. **订单防重**：防止重复支付
3. **金额验证**：回调金额与订单金额必须一致
4. **异步通知**：支付成功后必须异步处理
5. **敏感信息**：支付密钥不要提交到版本控制
